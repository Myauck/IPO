001import java.util.HashMap;
002
003/**
004 * Class Room - a room in an adventure game.
005 *
006 * This class is part of the "World of Zuul" application. 
007 * "World of Zuul" is a very simple, text based adventure game.  
008 *
009 * A "Room" represents one location in the scenery of the game.  It is 
010 * connected to other rooms via exits.  For each existing exit, the room 
011 * stores a reference to the neighboring room.
012 * 
013 * @author  Michael Kolling, David J. Barnes, Leo Gaillet
014 * @version 1.0 (February 2002) DBMOD:04/04/2008, 2019, 2023
015 */
016public class Room
017{
018    
019    private final String description;
020    private final HashMap<String, Room> exits = new HashMap<String, Room>();
021    private final String imageName;
022    private final ItemList items;
023    
024    
025    /**
026     * Constructeur de la classe Room
027     * Permet d'initialiser une nouvelle salle
028     * @param description Description de la salle instanciee
029     * @param imageName Nom de l'image
030     */
031    public Room(final String description, final String imageName) {
032        this.description = description;
033        this.imageName = imageName;
034        this.items = new ItemList();
035    }
036    
037    /**
038     * Getter qui recupere la description de la salle
039     * @return Description de la salle
040     */
041    public String getDescription() {
042        return this.description;
043    }
044 
045    /**
046     * Getter qui permet de recuperer le nom de l'image
047     * @return Nom de l'image
048     */
049    public String getImageName() {
050        return this.imageName;
051    }
052    
053    /**
054     * Getter qui recupere la salle d'une sortie en fonction de la direction de la sortie
055     * @param direction Direction de la sortie qu'on cherche
056     * @return Salle se trouvant a la sortie recherchee
057     */
058    public Room getExit(final String direction) {
059        return this.exits.get(direction.toLowerCase());
060    }
061    
062    /**
063     * Setter qui affecte une nouvelle salle de sortie en fonction de la direction que l'on a choisie
064     * @param direction Direction de la sortie
065     * @param exitRoom Salle qui se trouve dans la direction choisie
066     */
067    public void setExit(final String direction, final Room exitRoom) {
068        if (exitRoom != null)
069            this.exits.put(direction.toLowerCase(), exitRoom);
070    }
071    
072    /**
073     * Getter qui recupere les sorties disponibles dans la salle instanciee
074     * @return Chaine comportant l'ensemble des sorties disponibles
075     */
076    public String getExitsString() {
077        String availableExits = "Available Exits : ";
078        for(String keys : this.exits.keySet())
079            availableExits += keys + " ";
080        return availableExits;
081    }
082
083    /**
084     * Getter qui permet de recuperer l'integralite des informations de la salle
085     * @return Description complete de la salle instanciee ainsi que la presence des items
086     */
087    public String getLongDescription() {
088        return getDescription() + "\n" + getExitsString() + "\n" + getItemString();
089    }
090    
091    /**
092     * Récupère la liste des Items disponible dans la salle
093     * @return ItemList de la salle
094     */
095    public ItemList getItemList() {
096        return this.items;
097    }
098    
099    public String lookForItem(final String itemName) {
100        return this.items.getItem(itemName) != null ? this.items.getItem(itemName).getLongDescription() : "Item not found";
101    }
102    
103    /**
104     * Getter qui recupere l'ensemble des informations des items dans la salle
105     * @return Liste des descriptions des items dans la salle
106     */
107    public String getItemString() {
108        if(items.isEmpty())
109            return "No items here";
110        
111        StringBuilder itemContent = new StringBuilder();
112        
113        for(Item item : items.getContent()) {
114            itemContent.append("\"" + item.getName() + "\": ");
115            itemContent.append("\t" + item.getLongDescription() + "\n");
116        }
117        
118        return "Available items (" + items.getSize() + ") : " + itemContent.toString();
119    }
120}
