123    /**

126    private void printHelp() {
127        this.userInterface.println("Your commands are : " + parser.getCommandString());
128    }

131    /**
132     * Affiche la piece actuelle et ses sorties
133     */
134    private void printLocationInfo() {
135        this.userInterface.println(this.currentRoom.getLongDescription());
136        if(this.currentRoom.getImageName() != null)
137            this.userInterface.showImage(currentRoom.getImageName());
138    }

141    /**
142     * Permet d'acceder a une piece selon la direction et affiche la salle ou nous sommes
143     * @param command Direction dans laquelle le joueur souhaite aller
144     */
145    private void goRoom(final Command command) {
146        if (!command.hasSecondWord()){
147            this.userInterface.println("Go where ?");
148            return;
149        }
150        
151        Room nextRoom = this.currentRoom.getExit(command.getSecondWord());
152        
153        if (nextRoom == null){
154            this.userInterface.println("There is no door !");
155            return;
156        }
157        
158        this.previousRooms.push(this.currentRoom);
159        this.currentRoom = nextRoom;
160        
161        printLocationInfo();
162    }

165    /**
166     * Permet de quitter le jeu
167     * @param command Commande ecrite par l'utilisateur
168     * @return Vraie si le joueur ecrit quitter ou faux si il ya un second mot
169     */
170    private boolean quit(final Command command) {
171        if (command.hasSecondWord()){
172            this.userInterface.println("Quit what?");
173            return false;
174        }else{
175            this.userInterface.println("Thank you for playing.  Good bye.");
176            this.userInterface.enable(false);
177            return true;
178        }// else
179    } 
180    
181    /**
182     * Permet d'afficher la description complete de la salle dans laquelle nous sommes actuellement
183     * @param command Commande look
184     */
185    private void look(Command command) {
186        if(command.hasSecondWord()) {
187            String commandWord = command.getSecondWord();
188            if(this.currentRoom.getItem(commandWord) != null)
189                this.userInterface.println(this.currentRoom.getItem(commandWord).getLongDescription());
190            else
191                this.userInterface.println("Item not found");
192        }
193        else
194            this.userInterface.println(this.currentRoom.getLongDescription());
195    }

204    /**
205     * Permet d'executer une commande saisie par l'utilisateur
206     * @param rawCommand Commande que l'utilisateur a saisie
207     */
208    public void interpretCommand(final String rawCommand) {
209        this.userInterface.println( "> " + rawCommand );
210        Command command = this.parser.getCommand( rawCommand.toLowerCase() );
211
212        if (command.isUnknown()) {
213            this.userInterface.println("I don't know what you mean ...");
214            return;
215        }
216        
217        switch(command.getCommandWord().toLowerCase()) {
218            case "go":
219                goRoom(command);
220                return;
221            case "quit":
222                quit(command);
223                return;
224            case "help":
225                printHelp();
226                return;
227            case "look":
228                look(command);
229                return;
230            case "eat":
231                eat();
232                return;
233            case "back":
234                back(command);
235                return;
236            default:
237                this.userInterface.println("Unknown command !");
238                return;
239        }
240    }
241    
242    /**
243     * Permet de retourner dans la salle precedente
244     * @param command Commande back
245     */
246    public void back(final Command command) {
247        if(command.hasSecondWord()) {
248            this.userInterface.println("Back what ?");
249            return;
250        }
251        
252        if(this.previousRooms.size() == 0 || this.previousRooms.peek() == null) {
253            this.userInterface.println("There is any previous room !");
254            return;
255        }
256        
257        this.currentRoom = this.previousRooms.pop();
258        printLocationInfo();
259    }

264     */
265    public void setGUI(final UserInterface userInterface) {
266        this.userInterface = userInterface;
