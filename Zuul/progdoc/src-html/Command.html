001/**
002 * Classe Command - une commande du jeu d'aventure Zuul.
003 *
004 * @author Gaillet L
005 */
006public class Command {
007
008    private String commandWord;
009    private String secondWord;
010    
011    /**
012     * Constructeur de la classe Command
013     * @param commandWord Commande principale saisie par l'utilisateur
014     * @param secondWord Commande secondaire saisie par l'utilisateur
015     */
016    public Command (final String commandWord, final String secondWord) {
017        this.commandWord = commandWord;
018        this.secondWord = secondWord;
019    }
020    
021    /**
022     * Getter qui recupere la commande principale
023     * 
024     * @return Commande principale
025     */
026    public String getCommandWord () {
027        return this.commandWord;
028    }
029    
030    /**
031     * Getter qui recupere la commande secondaire
032     * @return Commande secondaire
033     */
034    public String getSecondWord () {
035        return this.secondWord;
036    }
037
038    /**
039     * Permet de savoir si la commande possede une commande secondaire
040     * @return Si la commande secondaire existe
041     */
042    public boolean hasSecondWord () {
043        return this.getSecondWord() != null;
044    }
045    
046    /**
047     * Permet de savoir si la commande principale n'est pas nulle
048     * @return Si la commande est nulle
049     */
050    public boolean isUnknown ()
051    {
052        return this.getCommandWord() == null;
053    }
054}
