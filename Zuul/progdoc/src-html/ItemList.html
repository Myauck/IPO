001import java.util.HashMap;
002
003/**
004 * ItemList est une classe dédiée à la gestion d'une collection des
005 * Item présent pour dans un joueur sous forme d'inventaire ou 
006 * sous forme d'un ensemble de fonctionnalités permettant
007 * de manipuler les Items qui se trouvent à l'interieur de celle-ci
008 *
009 * @author Leo GAILLET
010 * @version 20/04/2023
011 */
012public class ItemList {
013    
014    private final HashMap<String, Item> items = new HashMap<String, Item>();
015    
016    /**
017     * Permet de récupérer un item dans la liste des items
018     * @param itemName Nom de l'item
019     * @return Item s'il est trouvé
020     */
021    public Item getItem(final String itemName) {
022        return items.get(itemName.toLowerCase());
023    }
024    
025    /**
026     * Permet d'ajouter un item à la liste des items
027     * @param item Nouvel item à ajouter
028     */
029    public void addItem(final Item item) {
030        this.items.put(item.getName().toLowerCase(), item);
031    }
032    
033    /**
034     * Permet de retirer un item de la liste des items
035     * @param item que l'on souhaite retirer
036     */
037    public void removeItem(final Item item) {
038        this.items.remove(item.getName().toLowerCase());
039    }
040    
041    /**
042     * Permet de savoir s'il existe un item qui correspond
043     * @param itemName Item que l'on cherche
044     * @return Vrai s'il est trouvé, faux s'il ne l'est pas
045     */
046    public boolean hasItem(final String itemName) {
047        return this.items.containsKey(itemName.toLowerCase());
048    }
049    
050    /**
051     * Permet de savoir si la liste des items est vide
052     * @return Si la liste est vide
053     */
054    public boolean isEmpty() {
055        return getSize() == 0;
056    }
057    
058    /**
059     * Permet de récupérer la taille actuelle de la liste
060     * (Le nombre d'items disponible dans la liste d'items)
061     * @return Taille de la liste
062     */
063    public int getSize() {
064        return this.items.size();
065    }
066    
067    /**
068     * Permet de récupérer l'ensemble des items
069     * @return Tableau
070     */
071    public Item[] getContent() {
072        Item[] availableItems = new Item[items.size()];
073        int i = 0;
074        for(Item item : items.values()) {
075            availableItems[i] = item;
076            i++;
077        }
078        return availableItems;
079    }
080    
081}
